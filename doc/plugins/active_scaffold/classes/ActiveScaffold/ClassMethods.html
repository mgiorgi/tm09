active_scaffold_controller_for(klass)

Tries to find a controller for the given ActiveRecord model. Searches in the namespace of the current controller for singular and plural versions of the conventional "#{model}Controller" syntax. You may override this method to customize the search routine.

109:     def active_scaffold_controller_for(klass)
110:       namespace = self.to_s.split('::')[0...-1].join('::') + '::'
111:       error_message = []
112:       ["#{klass.to_s.underscore.pluralize}", "#{klass.to_s.underscore.pluralize.singularize}"].each do |controller_name|
113:         begin
114:           controller = "#{namespace}#{controller_name.camelize}Controller".constantize
115:         rescue NameError => error
116:           # Only rescue NameError associated with the controller constant not existing - not other compile errors
117:           if error.message["uninitialized constant #{controller}"]
118:             error_message << "#{namespace}#{controller_name.camelize}Controller"
119:             next
120:           else
121:             raise
122:           end
123:         end
124:         raise ActiveScaffold::ControllerNotFound, "#{controller} missing ActiveScaffold", caller unless controller.uses_active_scaffold?
125:         raise ActiveScaffold::ControllerNotFound, "ActiveScaffold on #{controller} is not for #{klass} model.", caller unless controller.active_scaffold_config.model == klass
126:         return controller
127:       end
128:       raise ActiveScaffold::ControllerNotFound, "Could not find " + error_message.join(" or "), caller
